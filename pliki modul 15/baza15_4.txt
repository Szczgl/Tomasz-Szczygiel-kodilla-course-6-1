SELECT I.SUMMARY, J.NAME
FROM ISSUES I
JOIN ISSUESLISTS J ON I.ID_ISSUESLISTS = J.ID;

SELECT I.SUMMARY, I.DESCRIPTION, J.FIRSTNAME, J.LASTNAME
FROM ISSUES I
JOIN USERS J ON I.USER_ID_ASSIGNEDTO = J.ID;

SELECT J.FIRSTNAME, J.LASTNAME, COUNT(*) AS ILOSC_ZADAN
FROM USERS J
JOIN ISSUES AS I ON J.ID = I.USER_ID_ASSIGNEDTO
GROUP BY J.ID;
